          select * from (
            select  c.ROWID||':'||d.ROWID   "id",
            STVSUBJ_DESC      "subjectDescription",
            SCBCRSE_CRSE_NUMB "courseNumber",
            SCBCRSE_EFF_TERM  "termEffective",
            SCBCRSE_TITLE     "courseTitle",
            SCBDESC_TEXT_NARRATIVE "courseText",
            SCBCRSE_SUBJ_CODE      "subjectCode"
            from SCBCRSE c, STVSUBJ s, SCBDESC d
            where c.SCBCRSE_SUBJ_CODE=s.STVSUBJ_CODE
              and d.SCBDESC_SUBJ_CODE(+)=c.SCBCRSE_SUBJ_CODE
              and d.SCBDESC_CRSE_NUMB(+)=c.SCBCRSE_CRSE_NUMB
              and d.SCBDESC_TERM_CODE_EFF(+)=c.SCBCRSE_EFF_TERM
              and d.SCBDESC_TERM_CODE_END is null
              and c.SCBCRSE_CRSE_NUMB=nvl(:courseNumber,SCBCRSE_CRSE_NUMB)
              and c.SCBCRSE_EFF_TERM=nvl(:termEffective,SCBCRSE_EFF_TERM)
              and (substr(:id,1, instr(:id,':')-1) is null or c.ROWID=substr(:id,1, instr(:id,':')-1) )
              and ( substr(:id,instr(:id,':')+1)   is null or d.ROWID=substr(:id,instr(:id,':')+1) )
              --and SCBCRSE_SUBJ_CODE like 'AS%'
            order by SCBCRSE_SUBJ_CODE, SCBCRSE_CRSE_NUMB,SCBCRSE_EFF_TERM desc
         ) where rownum< 5
