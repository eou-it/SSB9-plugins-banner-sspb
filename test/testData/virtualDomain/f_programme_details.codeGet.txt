select 
rowidtochar(smrprle.rowid) "id"
,stvlevl_desc
,stvcamp_desc
,stvcoll_desc
,stvdegc_desc
,smrprle.*
from smrprle ,stvlevl , stvcamp, stvcoll, stvdegc
where stvlevl_code=smrprle_levl_code
  and stvcamp_code(+)=smrprle_camp_code
  and stvcoll_code(+)=smrprle_coll_code
  and stvdegc_code(+)=smrprle_degc_code
  and smrprle_program like nvl(:programme,'%')
